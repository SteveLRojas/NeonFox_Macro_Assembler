Notes on NeonFox architecture.

Registers:
$00 - $0F	R0 - R15
$10 - $1F
	$10 (16)	AUX0	IMPLICIT SOURCE
	$11 (17)	AUX1	IMPLICIT DESTINATION
	$12 (18)	AUX2
	$13 (19)	AUX3
	$14 (20)	DAL		DATA ADDRESS LOW
	$15 (21)	DAH		DATA ADDRESS HIGH
	$16 (22)	IAL		IO ADDRESS LOW
	$17 (23)	RESERVED	IO ADDRESS HIGH
	$18 (24)	DD		DATA SPACE DATA BUS
	$19 (25)	ID		IO SPACE DATA BUS
	$1A (26)	CAL		CALL ADDRESS LOW
	$1B (27)	CAH		CALL ADDRESS HIGH
		DETERMINES THE ADDRESS FOR THE NEXT RETX, RETL, CALLX OR CALLL INSTRUCTION.
	$1C (28)	RAL		RETURN ADDRESS LOW
	$1D (29)	RAH		RETURN ADDRESS HIGH
		CONTAINS THE RETURN ADDRESS FOR THE LAST CALLX INSTRUCTION.
	$1E (30)	STATUS
	$1F (31)	ZERO	ALWAYS ZERO

Instructions:
	ADD
	ADDC
	SUB
	SUBC
	MOVE, TEST
	NOT
	ROR
	ROL
	AND
	XOR
	OR
	CALL, CALLX, CALLL, CALLLX, RET, RETX, RETL, RETLX	NOTE: CALLX = RETX, CALLLX = RETLX, ALIAS TO JMP AND JMPL FOR CODE READABILITY.
	NOP, BRZ, BRN, BRP
	BRA, BRNZ, BRNN, BRNP
	LIM
	BITT

Instruction examples:
	ADD W, R1, R2
	ADD H, R1, R2
	ADD L, R1, R2
	ADD S, R1, R2
	ADDC W, R1, R2
	SUB W, R1, R2		(R1 - AUX0 -> R2)
	SUBC W, R1, R2
	MOVE W, R1, R2
	MOVE H, R1, R2
	MOVE L, R1, R2
	MOVE S, R1, R2
	TEST W, R1
	TEST H, R1
	TEST L, R1
	TEST S, R1
	NOT W, R1, R2
	ROR W, R1, R2
	ROL W, R1, R2
	AND W, R1, R2
	XOR W, R1, R2
	OR W, R1, R2
	CALL
	CALLX
	CALLL W, $FF
	CALLL H, $FF
	CALLL L, $FF
	CALLLX W, $FF
	CALLLX H, $FF
	CALLLX L, $FF
	RET
	RETX
	RETL W, $FF
	RETL H, $FF
	RETL L, $FF
	RETLX W, $FF
	RETLX H, $FF
	RETLX L, $FF
	JMP
	JMPL W, $FF
	JMPL H, $FF
	JMPL L, $FF
	NOP
	BRZ LBL
	BRN LBL
	BRP LBL
	BRA LBL
	BRNZ LBL
	BRNN LBL
	BRNP LBL
	LIM W, AUX0, $FA
	LIM H, AUX1, $FA
	LIM L, AUX2, $FA
	LIM L, AUX3, $FA
	BITT W, R4, 15
	BITT H, R4, 7
	BITT L, R4, 6
